f_tf_to_minutes(string tf_str) => timeframe.in_seconds(tf_str) / 60
f_format_tf_label(string tf_str) =>
    mins = f_tf_to_minutes(tf_str)
    mins >= 60 and mins % 60 == 0 ? str.tostring(mins / 60) + "H" : str.tostring(mins) + "m"
f_is_entry(string tf_str) =>
    mins = f_tf_to_minutes(tf_str)
    mins >= 1 and mins <= 5
f_get_full_category(string tf_str) =>
    tf_minutes = f_tf_to_minutes(tf_str)
    if tf_minutes >= 1 and tf_minutes <= 5
        "ENTRY"
    else if tf_minutes >= 6 and tf_minutes <= 20
        "SCALP"
    else if tf_minutes >= 30 and tf_minutes <= 120
        "INTRA"
    else
        "NONE"
f_pattern_str(bool is_bear, bool third, bool first, bool laol, bool sn, bool sn_dbl, bool fu, bool tbe, bool hcs, bool hcs_forming) =>
    string result = ""

    if first
        result := result == "" ? "X3 First" : result + " + X3 First"
    if laol
        result := result == "" ? "LAOL Neg" : result + " + LAOL Neg"
    if sn
        sn_txt = sn_dbl ? "SN [EM]" : "SN"
        result := result == "" ? sn_txt : result + " + " + sn_txt
    if fu
        fu_txt = tbe ? "FU [TBE✓]" : "FU"
        result := result == "" ? fu_txt : result + " + " + fu_txt
    if hcs
        result := result == "" ? "[HCS ✓]" : result + " + [HCS ✓]"
    else if hcs_forming
        result := result == "" ? "[HCS]" : result + " + [HCS]"
    result != "" ? (is_bear ? "Bear " : "Bull ") + result : ""

f_is_fu_pattern(string pattern) =>
    if pattern == ""
        false
    else
        has_fu = str.contains(pattern, "FU")
        has_sn = str.contains(pattern, "SN")
        has_em_modifier = str.contains(pattern, "HCS") or str.contains(pattern, "Third") or str.contains(pattern, "First") or str.contains(pattern, "LAOL") or str.contains(pattern, "TBE")
        (has_fu or has_sn) and not has_em_modifier
f_is_em_pattern(string pattern) =>
    if pattern == ""
        false
    else
        str.contains(pattern, "HCS") or str.contains(pattern, "Third") or str.contains(pattern, "First") or str.contains(pattern, "LAOL") or str.contains(pattern, "TBE") or str.contains(pattern, "[EM]")
f_has_both_wicks(float o, float h, float l, float c) =>

f_bullish_ib_confirmation(float p_o1, float p_h1, float p_l1, float p_c1) =>

f_update_seq(seq_state seq, bool is_fu, bool is_x3, bool counter_fu, bool counter_x3, float level, float body, string tf_str, bool confirmed, bool is_bear, float p_h, float p_l) =>
    if not confirmed
        [seq.step, false]
    else
        tf_seconds = timeframe.in_seconds(tf_str)
        candles_since = not na(seq.start_time) ? math.round((time - seq.start_time) / (tf_seconds * 1000)) : 0
        if seq.step > 0 and seq.step < 5 and candles_since > 5
            seq.step := 0
            seq.level := na
            seq.body := na
            seq.start_time := na
        is_valid = false
        if seq.step == 5
            seq.step := 0
            seq.level := na
            seq.body := na
            seq.start_time := na
        else if seq.step == 4
            broke_level = is_bear ? (p_h > seq.level) : (p_l < seq.level)
            if broke_level
                seq.step := 0
                seq.level := na
                seq.body := na
                seq.start_time := na
            else if is_fu
                seq.step := 5
                is_valid := true
        else if seq.step == 3
            broke_level = is_bear ? (p_h > seq.level) : (p_l < seq.level)
            in_zone = is_bear ? (p_h > seq.body and p_h < seq.level) : (p_l < seq.body and p_l > seq.level)
            if broke_level
                seq.step := 0
                seq.level := na
                seq.body := na
                seq.start_time := na
            else if in_zone
                seq.step := 4
        else if seq.step == 2
            broke_level = is_bear ? (p_h > seq.level) : (p_l < seq.level)
            if broke_level
                seq.step := 0
                seq.level := na
                seq.body := na
                seq.start_time := na
            else if counter_fu or counter_x3
                seq.step := 3
        else if seq.step == 1
            broke_level = is_bear ? (p_h > seq.level) : (p_l < seq.level)
            in_zone = is_bear ? (p_h >= seq.body and p_h <= seq.level) : (p_l <= seq.body and p_l >= seq.level)
            if broke_level
                seq.step := 0
                seq.level := na
                seq.body := na
                seq.start_time := na
            else if in_zone
                seq.step := 2
            else
                seq.step := 0
                seq.level := na
                seq.body := na
                seq.start_time := na
        if (is_fu or is_x3) and seq.step == 0
            seq.step := 1
            seq.level := level
            seq.body := body
            seq.start_time := time
        [seq.step, is_valid]
f_time_ago(int past_time) =>
    if past_time == 0
        ""
    else
        elapsed_ms = time - past_time
        elapsed_min = math.floor(elapsed_ms / 60000)
        if elapsed_min < 1
            " (<1m ago)"
        else if elapsed_min < 60
            " (" + str.tostring(elapsed_min) + "m ago)"
        else if elapsed_min < 1440
            hours = math.floor(elapsed_min / 60)
            mins = elapsed_min % 60
            " (" + str.tostring(hours) + "h" + (mins > 0 ? str.tostring(mins) + "m" : "") + " ago)"
        else
            days = math.floor(elapsed_min / 1440)
            " (" + str.tostring(days) + "d ago)"
f_push_unique(string[] arr, string txt) =>
    bool found = false
    for existing in arr
        if existing == txt
            found := true
            break
    if not found
        array.push(arr, txt)
    0
f_check_break(last_valid_info lv, float h, float l) =>
    if lv.original_text != "None" and not na(lv.level)
        if lv.direction == "bear" and h > lv.level
            lv.pattern_text := "BROKEN (was " + lv.original_text + ")"
            lv.is_broken := true
        if lv.direction == "bull" and l < lv.level
            lv.pattern_text := "BROKEN (was " + lv.original_text + ")"
            lv.is_broken := true
    0
f_resolve_last_valid(last_valid_info bear_lv, last_valid_info bull_lv) =>
    string lv_text = "None"
    string lv_dir = ""
    string lv_orig_dir = ""
    bool lv_broken = false
    if bear_lv.est_time > 0 or bull_lv.est_time > 0
        if bear_lv.est_time > bull_lv.est_time
            lv_text := bear_lv.pattern_text + f_time_ago(bear_lv.est_time)
            lv_dir := bear_lv.is_broken ? "bull" : "bear"
            lv_orig_dir := "bear"
            lv_broken := bear_lv.is_broken
        else
            lv_text := bull_lv.pattern_text + f_time_ago(bull_lv.est_time)
            lv_dir := bull_lv.is_broken ? "bear" : "bull"
            lv_orig_dir := "bull"
            lv_broken := bull_lv.is_broken
    [lv_text, lv_dir, lv_orig_dir, lv_broken]

f_add_or_merge_laol_line(laol_line_data[] lines, float level, string tf_label, bool is_bear, string category) =>
    tolerance = syminfo.mintick * 2
    merged = false
    bool found_broken = false
    if array.size(lines) > 0
        for laol in lines
            if math.abs(laol.level - level) < tolerance
                if laol.is_broken
                    found_broken := true
                    break
                if not str.contains(laol.tf_labels, tf_label)
                    laol.tf_labels := laol.tf_labels + "," + tf_label
                    laol.tf_count  := laol.tf_count + 1
                    if category == "INTRA"
                        laol.has_intra := true
                    else if category == "SCALP"
                        laol.has_scalp := true
                    else
                        laol.has_entry := true
                merged := true
                break
    if not merged and not found_broken
        new_laol = laol_line_data.new()
        new_laol.level      := level
        new_laol.tf_labels  := tf_label
        new_laol.creation_bar := bar_index
        new_laol.is_bear    := is_bear
        new_laol.tf_count   := 1
        new_laol.has_entry  := category == "ENTRY"
        new_laol.has_scalp  := category == "SCALP"
        new_laol.has_intra  := category == "INTRA"
        array.push(lines, new_laol)
    0
f_update_laol_lines(laol_line_data[] lines, bool is_bear, float current_price) =>
    int break_time = 0
    string break_tf = ""
    int intra_break_time = 0
    string intra_break_tf = ""
    int scalp_break_time = 0
    string scalp_break_tf = ""
    if array.size(lines) > 0
        for idx = array.size(lines) - 1 to 0
            if idx >= array.size(lines)
                continue
            laol = array.get(lines, idx)
            crossed = is_bear ? (current_price > laol.level) : (current_price < laol.level)
            if crossed and not laol.is_broken
                laol.is_broken  := true
                laol.break_bar  := bar_index
                if laol.has_intra
                    intra_break_time := time
                    intra_break_tf   := laol.tf_labels
                else if laol.has_scalp
                    scalp_break_time := time
                    scalp_break_tf   := laol.tf_labels
                else
                    break_time := time
                    break_tf   := laol.tf_labels
            if laol.is_broken and bar_index >= laol.break_bar + LAOL_DELETE_DELAY
                array.remove(lines, idx)
    [break_time, break_tf, intra_break_time, intra_break_tf, scalp_break_time, scalp_break_tf]
f_get_broken_laol(laol_line_data[] lines, bool require_multi) =>
    float result_level = na
    int result_tf_count = 0
    if array.size(lines) > 0
        for laol in lines
            if laol.is_broken and bar_index == laol.break_bar
                if not require_multi or laol.tf_count >= 2
                    result_level := laol.level
                    result_tf_count := laol.tf_count
                    break
    [result_level, result_tf_count]

f_scan_laol_in_box(laol_line_data[] laol_arr, float _top, float _bot, bool _is_bear, string _laol_cat, string _box_cat) =>
    float _tol  = syminfo.mintick * 3
    for _ll in laol_arr
        if _ll.is_broken
            continue
        if _ll.level <= _top and _ll.level >= _bot
            bool _already = false
            int  _sz      = array.size(_xlaol_levels)
            if _sz > 0
                for _ei = 0 to _sz - 1
                    if math.abs(array.get(_xlaol_levels, _ei) - _ll.level) < _tol and array.get(_xlaol_cat, _ei) == _laol_cat
                        _already := true
                        break
            if not _already
                array.push(_xlaol_levels, _ll.level)
                array.push(_xlaol_bear,   _is_bear)
                array.push(_xlaol_cat,    _laol_cat)
                array.push(_xlaol_off,    0)
    0

f_create_box(tracked_box[] boxes, string direction, float top_v, float bottom_v, int creation_t, string pattern, string tf_str, bool is_intra, bool should_show) =>
    bool result = false
    if pattern != "" and not na(top_v) and not na(bottom_v) and not na(creation_t)
        exists = false
        for bx in boxes
            if bx.creation_time == creation_t and bx.direction == direction and bx.timeframe == tf_str
                exists := true
                break
        if not exists
            new_box = tracked_box.new()
            new_box.direction      := direction
            new_box.state          := "FORMING"
            new_box.top_val        := top_v
            new_box.bottom_val     := bottom_v
            new_box.original_top   := top_v
            new_box.original_bottom := bottom_v
            new_box.creation_time  := creation_t
            new_box.pattern_text   := pattern
            new_box.base_pattern   := pattern
            new_box.timeframe      := tf_str
            new_box.is_intra       := is_intra
            bool is_em = f_is_em_pattern(pattern)
            new_box.is_em_forming  := is_em
            array.unshift(boxes, new_box)
            result := true
    result
f_manage_hcs_boxes(hcs_box_data[] hcs_boxes, bool is_bear) =>
    bool any_retesting = false
    if array.size(hcs_boxes) > 0
        for idx = array.size(hcs_boxes) - 1 to 0
            if idx >= array.size(hcs_boxes)
                continue
            hcs = array.get(hcs_boxes, idx)
            if is_bear
                if not hcs.is_broken and high > hcs.top_val
                    hcs.is_broken := true
                if hcs.is_broken
                    if high >= hcs.bottom_val and high <= hcs.top_val
                        any_retesting := true
                    if low < hcs.bottom_val
                        array.remove(hcs_boxes, idx)
                        continue
                    if low <= hcs.top_val and low >= hcs.bottom_val
                        if low < hcs.top_val
                            hcs.top_val := low
            else
                if not hcs.is_broken and low < hcs.bottom_val
                    hcs.is_broken := true
                if hcs.is_broken
                    if low >= hcs.bottom_val and low <= hcs.top_val
                        any_retesting := true
                    if high > hcs.top_val
                        array.remove(hcs_boxes, idx)
                        continue
                    if low >= hcs.bottom_val and low <= hcs.top_val
                        if low > hcs.bottom_val
                            hcs.bottom_val := low
    any_retesting

    if array.size(items) == 0
        items

        string[] combined = array.new_string()
        for item in items
            tf_start = str.pos(item, "[")
            tf_end = str.pos(item, "]")
            if tf_start >= 0 and tf_end > tf_start
                tf_label = str.substring(item, tf_start, tf_end + 1)
                is_bear = str.contains(item, "Bear")
                bool found = false
                if array.size(combined) > 0
                    for ci = 0 to array.size(combined) - 1
                        existing = array.get(combined, ci)
                        if str.contains(existing, tf_label)
                            if str.contains(existing, "Bear") == is_bear
                                pattern_start = tf_end + 2
                                new_pattern = str.substring(item, pattern_start, str.length(item))
                                array.set(combined, ci, existing + " + " + new_pattern)
                                found := true
                                break
                if not found
                    array.push(combined, item)
            else
                array.push(combined, item)
        combined
